269    def _calculate_frame_sizes(self) -> list[FrameEntry]:
270        """
271        Get the frame sizes via ffprobe using the -show_packets option.
272        This includes the NAL headers, of course.
273
274        Returns:
275            list[dict]: The frame sizes plus some extra info.
276        """
277        logger.debug(f"Calculating frame size from {self.input_file}")
278
279        base_cmd = [
280            "ffprobe",
281            "-loglevel",
282            "error",
283            "-select_streams",
284            self.stream_type[0] + ":0",
285        ]
286
287        if self.read_length:
288            base_cmd.extend(["-read_intervals", self.read_length])
289
290        base_cmd.extend(
291            [
292                "-show_packets",
293                "-show_entries",
294                "packet=pts_time,dts_time,duration_time,size,flags",
295                "-of",
296                "json",
297                self.input_file,
298            ]
299        )
300
301        # Get frame count for progress bar (only if showing progress)
302        frame_count = None
303        if self.show_progress:
304            frame_count = self._get_frame_count()
305
306        stdout, _ = run_command(
307            base_cmd,
308            self.dry_run,
309            show_progress=self.show_progress,
310            total_frames=frame_count,
311        )
312        if self.dry_run or stdout is None:
313            logger.error("Aborting prematurely, dry-run specified or stdout was empty")
314            sys.exit(0)
315
316        info = json.loads(stdout)["packets"]
317        # abort if nothing returned
318        if not info:
319            raise RuntimeError(
320                "No frames returned, please check your input file and the ffprobe output. Run with -v to see the ffprobe command."
321            )
322
323        ret: list[FrameEntry] = []
324        idx = 1
325
326        default_duration = next(
327            (x["duration_time"] for x in info if "duration_time" in x.keys()), "NaN"
328        )
329
330        for packet_info in info:
331            frame_type: Literal["I", "Non-I"] = (
332                # key frames are marked with a capital K (K_ or K__) in packet flags
333                "I" if "K" in packet_info["flags"] else "Non-I"
334            )
335
336            pts: float | Literal["NaN"] = (
337                float(packet_info["pts_time"])
338                if "pts_time" in packet_info.keys()
339                else "NaN"
340            )
341
342            duration: float | Literal["NaN"] = (
343                float(packet_info["duration_time"])
344                if "duration_time" in packet_info.keys()
345                else float(default_duration)
346                if default_duration != "NaN"
347                else "NaN"
348            )
349
350            ret.append(
351                {
352                    "n": idx,
353                    "frame_type": frame_type,
354                    "pts": pts,
355                    "size": int(packet_info["size"]),
356                    "duration": duration,
357                }
358            )
359            idx += 1
360
361        # fix for missing durations, estimate it via PTS
362        if default_duration == "NaN":
363            ret = self._fix_durations(ret)
364
365        # fix missing data in first packet (occurs occassionally when reading streams)
366        if ret[0]["duration"] == "NaN" and isinstance(ret[1]["duration"], float):
367            ret[0]["duration"] = ret[1]["duration"]
368
369        if (
370            ret[0]["pts"] == "NaN"
371            and isinstance(ret[1]["pts"], float)
372            and isinstance(ret[0]["duration"], float)
373        ):
374            ret[0]["pts"] = ret[1]["pts"] - ret[0]["duration"]
375
376        self.frames = ret
377        return ret

379    def _fix_durations(self, ret: List[FrameEntry]) -> List[FrameEntry]:
380        """
381        Calculate durations based on delta PTS.
382        """
383        last_duration = None
384        for i in range(len(ret) - 1):
385            curr_pts = ret[i]["pts"]
386            next_pts = ret[i + 1]["pts"]
387            if curr_pts == "NaN" or next_pts == "NaN":
388                logger.warning("PTS is NaN, duration/bitrate may be invalid")
389                continue
390            if next_pts < curr_pts:
391                logger.warning(
392                    "Non-monotonically increasing PTS, duration/bitrate may be invalid"
393                )
394            last_duration = next_pts - curr_pts
395            ret[i]["duration"] = last_duration
396        if last_duration is not None:
397            ret[-1]["duration"] = last_duration
398        return ret

422    def _collect_chunks(self) -> list[float]:
423        """
424        Collect chunks of a certain aggregation length (in seconds, or GOP).
425        This is cached.
426
427        Returns:
428            list[float]: The bitrate values per chunk in kbit/s.
429        """
430        if len(self._chunks):
431            return self._chunks
432
433        logger.debug("Collecting chunks for bitrate calculation")
434
435        # this is where we will store the stats in buckets
436        aggregation_chunks: list[list[FrameEntry]] = []
437        curr_list: list[FrameEntry] = []
438
439        self._gop_start_times = []
440
441        if self.aggregation == "gop":
442            # collect group of pictures, each one containing all frames belonging to it
443            for frame in self.frames:
444                if frame["frame_type"] != "I":
445                    curr_list.append(frame)
446                if frame["frame_type"] == "I":
447                    if curr_list:
448                        aggregation_chunks.append(curr_list)
449                    curr_list = [frame]
450                    self._gop_start_times.append(float(frame["pts"]))
451            # flush the last one
452            aggregation_chunks.append(curr_list)
453
454        else:
455            # per-time aggregation
456            agg_time: float = 0
457            for frame in self.frames:
458                if agg_time < self.chunk_size:
459                    curr_list.append(frame)
460                    agg_time += float(frame["duration"])
461                else:
462                    if curr_list:
463                        aggregation_chunks.append(curr_list)
464                    curr_list = [frame]
465                    agg_time = float(frame["duration"])
466            aggregation_chunks.append(curr_list)
467
468        # calculate BR per group
469        self._chunks = [
470            BitrateStats._bitrate_for_frame_list(x) for x in aggregation_chunks
471        ]
472
473        return self._chunks

475    @staticmethod
476    def _bitrate_for_frame_list(frame_list: list[FrameEntry]) -> float:
477        """
478        Given a list of frames with size and PTS, get the bitrate.
479
480        Args:
481            frame_list (list): list of frames
482
483        Returns:
484            float: bitrate in kbit/s
485        """
486        if len(frame_list) < 2:
487            return math.nan
488
489        # sort frames by PTS in case they are unordered
490        frame_list.sort(key=lambda x: x["pts"])
491
492        duration = float(frame_list[-1]["pts"]) - float(frame_list[0]["pts"])
493        size = sum(f["size"] for f in frame_list)
494        bitrate = ((size * 8) / 1000) / duration
495
496        return bitrate

498    def _calculate_max_min_bitrate(self) -> tuple[float, float]:
499        """
500        Find the min/max from the chunks.
501
502        Returns:
503            tuple: max, min bitrate in kbit/s
504        """
505        self.max_bitrate = max(self._collect_chunks())
506        self.min_bitrate = min(self._collect_chunks())
507        return self.max_bitrate, self.min_bitrate

509    def _assemble_bitrate_statistics(self) -> BitrateStatsSummary:
510        """
511        Assemble all pre-calculated statistics plus some simple statistical measures.
512
513        Returns:
514            dict: bitrate statistics
515        """
516
517        self.avg_bitrate = (
518            sum(f["size"] for f in self.frames) * 8 / 1000
519        ) / self.duration
520        self.avg_bitrate_over_chunks: float = cast(
521            float, np.mean(self._collect_chunks())
522        )
523
524        self.max_bitrate_factor = self.max_bitrate / self.avg_bitrate
525
526        # output data
527        ret: BitrateStatsSummary = {
528            "input_file": self.input_file,
529            "stream_type": self.stream_type,
530            "avg_fps": round(self.fps, self.rounding_factor),
531            "num_frames": len(self.frames),
532            "avg_bitrate": round(self.avg_bitrate, self.rounding_factor),
533            "avg_bitrate_over_chunks": round(
534                self.avg_bitrate_over_chunks, self.rounding_factor
535            ),
536            "max_bitrate": round(self.max_bitrate, self.rounding_factor),
537            "min_bitrate": round(self.min_bitrate, self.rounding_factor),
538            "max_bitrate_factor": round(self.max_bitrate_factor, self.rounding_factor),
539            "bitrate_per_chunk": [
540                round(b, self.rounding_factor) for b in self._collect_chunks()
541            ],
542            "aggregation": self.aggregation,
543            "chunk_size": self.chunk_size if self.aggregation == "time" else None,
544            "duration": round(self.duration, self.rounding_factor),
545        }
546
547        self.bitrate_stats = ret
548        return self.bitrate_stats

612
613        fig = plotille.Figure()

615        fig.height = height
616
617        def _round_decimals(
618            val: Any, chars: int, delta: float, left: bool = False

620            align = "<" if left else ""
621            return f"{val:{align}{chars}.{self.rounding_factor}f}"
622
623        x_values = (
624            self._gop_start_times

628
629        fig.register_label_formatter(int, _round_decimals)

632        fig.set_y_limits(min_=0)

634        fig.y_label = "Bitrate (kbit/s)"
635
636        fig.plot(
637            x_values,
638            chunks,
639        )
640
641        print(fig.show(), file=sys.stderr)
